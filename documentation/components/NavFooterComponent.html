import { Component, OnInit, EventEmitter, Output, Input } from '@angular/core';


@Component({
  selector: 'nav-footer',
  templateUrl: './nav-footer.component.html',
  styleUrls: ['./nav-footer.component.css']
})
export class NavFooterComponent {

  constructor() { }

  @Output() forward: EventEmitter<any> = new EventEmitter();
  @Output() backward: EventEmitter<any> = new EventEmitter();

  @Input() forwardEnabled = true; 
  @Input() backwardEnabled = true;

     
  onForward(event) {
    this.forward.emit(event);
  }

  onBackward(event) {
    this.backward.emit(event);
  }

}

<div class="footer">
  <button mat-raised-button color="primary" class="footer-lbutton" type="button" [disabled]="!backwardEnabled" (click)="onBackward()">Назад</button>
  <button mat-raised-button color="primary" class="footer-rbutton" type="button" [disabled]="!forwardEnabled" (click)="onForward()">Вперед</button>
</div>
